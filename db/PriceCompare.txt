-- Création de la base de données
CREATE DATABASE PriceCompare;
USE PriceCompare;

-- Table pour les utilisateurs
CREATE TABLE UserEntity (
    UserID INT PRIMARY KEY IDENTITY(1,1),
    FirstName NVARCHAR(200) NOT NULL,
    LastName NVARCHAR(200) NOT NULL,
    City NVARCHAR(200) NOT NULL,
    BirthDate DATE NOT NULL,
    Mail NVARCHAR(200) UNIQUE NOT NULL,
    PasswordHash NVARCHAR(255) NOT NULL
);

-- Table pour les rôles des utilisateurs (super_manager, manager, user)
CREATE TABLE RoleEntity (
    RoleID INT PRIMARY KEY IDENTITY(1,1),
    RoleName NVARCHAR(50) UNIQUE NOT NULL
);

-- Table de liaison entre les utilisateurs et leurs rôles
CREATE TABLE RolesEntity (
    UserID INT NOT NULL,
    RoleID INT NOT NULL,
    FOREIGN KEY (UserID) REFERENCES UserEntity(UserID) ON DELETE CASCADE,
    FOREIGN KEY (RoleID) REFERENCES RoleEntity(RoleID) ON DELETE CASCADE,
    PRIMARY KEY (UserID, RoleID)
);

-- Table pour les magasins
CREATE TABLE StoreEntity (
    StoreID INT PRIMARY KEY IDENTITY(1,1),
    StoreName NVARCHAR(255) NOT NULL UNIQUE,
    StoreCity NVARCHAR(200) NOT NULL,
    StoreAddress NVARCHAR(255) NOT NULL
);

-- Table pour les unités des articles
CREATE TABLE UnitEntity (
    UnitID INT PRIMARY KEY IDENTITY(1,1),
    UnitName NVARCHAR(50) NOT NULL UNIQUE -- Exemple : "Kilo", "Litre", "Pièce"
);

-- Table pour les articles
CREATE TABLE ArticleEntity (
    ArticleID INT PRIMARY KEY IDENTITY(1,1),
    ArticleName NVARCHAR(255) NOT NULL,
    UnitID INT NOT NULL, -- Clé étrangère vers UnitEntity
    ArticleDefaultPrice DECIMAL(10, 2) NOT NULL,
    LastUpdated DATETIME DEFAULT GETDATE(),
    FOREIGN KEY (UnitID) REFERENCES UnitEntity(UnitID) ON DELETE CASCADE
);

-- Table pour les prix des articles
CREATE TABLE PriceEntity (
    PriceID INT PRIMARY KEY IDENTITY(1,1),
    StoreID INT NOT NULL,
    StartDate DATE NOT NULL,
    EndDate DATE,
    Price DECIMAL(10, 2) NOT NULL,
    PreviousPrice DECIMAL(10, 2),
    IsPromotion BIT DEFAULT 0,
    IsEstimatedPrice BIT DEFAULT 0,
    FOREIGN KEY (StoreID) REFERENCES StoreEntity(StoreID) ON DELETE CASCADE
);

-- Table de liaison entre les prix et les articles (catalogue)
CREATE TABLE catalogueEntity (
    PriceID INT NOT NULL,
    ArticleID INT NOT NULL,
    FOREIGN KEY (PriceID) REFERENCES PriceEntity(PriceID) ON DELETE CASCADE,
    FOREIGN KEY (ArticleID) REFERENCES ArticleEntity(ArticleID) ON DELETE CASCADE,
    PRIMARY KEY (PriceID, ArticleID) -- Clé primaire composite pour éviter les doublons
);

-- Table pour les rapports de nouveaux prix des utilisateurs
CREATE TABLE priceReportsEntity (
    ReportID INT PRIMARY KEY IDENTITY(1,1),
    ArticleID INT NOT NULL,
    StoreID INT NOT NULL,
    UserID INT NOT NULL,
    ReportedPrice DECIMAL(10, 2) NOT NULL,
    ReportDate DATE NOT NULL DEFAULT GETDATE(),
    ProofImage VARBINARY(MAX), -- Image ou scan de la preuve
    IsValidated BIT DEFAULT 0, -- Indique si un manager a validé le prix
    FOREIGN KEY (ArticleID) REFERENCES ArticleEntity(ArticleID) ON DELETE CASCADE,
    FOREIGN KEY (StoreID) REFERENCES StoreEntity(StoreID) ON DELETE CASCADE,
    FOREIGN KEY (UserID) REFERENCES UserEntity(UserID) ON DELETE CASCADE
);

-- Table pour les notifications de baisse de prix ou promotion
CREATE TABLE notificationsEntity (
    NotificationID INT PRIMARY KEY IDENTITY(1,1),
    UserID INT NOT NULL,
    ArticleID INT NOT NULL,
    StoreID INT NOT NULL,
    NotificationDate DATETIME DEFAULT GETDATE(),
    NotificationType NVARCHAR(50),
    IsRead BIT DEFAULT 0,
    NotificationMessage NVARCHAR(500),
    FOREIGN KEY (UserID) REFERENCES UserEntity(UserID) ON DELETE CASCADE,
    FOREIGN KEY (ArticleID) REFERENCES ArticleEntity(ArticleID) ON DELETE CASCADE,
    FOREIGN KEY (StoreID) REFERENCES StoreEntity(StoreID) ON DELETE CASCADE
);

-- Table pour les exports de données
CREATE TABLE exportsEntity (
    ExportID INT PRIMARY KEY IDENTITY(1,1),
    UserID INT NOT NULL,
    ExportDate DATETIME DEFAULT GETDATE(),
    ExportType NVARCHAR(50),
    ExportData VARBINARY(MAX), -- Contenu exporté
    ArticleID INT NOT NULL,  -- Relation avec un seul article
    FOREIGN KEY (UserID) REFERENCES UserEntity(UserID) ON DELETE CASCADE,
    FOREIGN KEY (ArticleID) REFERENCES ArticleEntity(ArticleID) ON DELETE CASCADE
);
